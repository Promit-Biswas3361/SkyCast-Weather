def classify_cr_from_excel_file_pandas(file_object: Union[TextIO, io.BytesIO, str], file_extension: str) -> dict:
    try:
        # Determine the file-like object or string to pass to pandas
        if isinstance(file_object, str):
            # If it's a string, assume it's the content and wrap in StringIO
            file_to_read = io.StringIO(file_object)
        elif isinstance(file_object, io.BytesIO):
            # For BytesIO, decode and wrap in StringIO
            file_to_read = io.StringIO(file_object.read().decode('utf-8'))
        elif isinstance(file_object, io.TextIOBase):
            # For TextIOBase, just use it directly
            file_to_read = file_object
        else:
            return {
                "status": "error",
                "error_message": f"Unexpected type for file_object: {type(file_object)}. Expected str, TextIO, or BytesIO."
            }

        if file_extension.lower() not in ("csv", "tsv", "txt"):
            return {
                "status": "error",
                "error_message": f"Unsupported file type: .{file_extension}. Only .csv, .tsv and .txt files are supported."
            }

        try:
            df = pd.read_csv(file_to_read, sep=None, engine='python')

            # Remove any entirely empty rows (where all values are NaN or whitespace)
            df.dropna(how='all', inplace=True)
            if df.empty:
                return {
                    "status": "error",
                    "error_message": "Processed file content is empty. No data to classify."
                }

        except pd.errors.EmptyDataError:
            return {
                "status": "error",
                "error_message": "Processed file content is empty. No data to classify."
            }
        except Exception as e:
            return {
                "status": "error",
                "error_message": f"Error reading file with pandas: {str(e)}"
            }

        original_fieldnames = list(df.columns)

        ticket_col = find_column(original_fieldnames, ['TicketNumber', 'Ticket Number', 'Ticket_Number', 'CR_ID', 'CR ID', 'ID'])
        implementer_col = find_column(original_fieldnames, ['ImplementerGroup', 'Implementer Group', 'Implementer_Group', 'ImplementerGroups'])
        deployment_col = find_column(original_fieldnames, ['Deployment Approach', 'DeploymentApproach', 'Deployment_Approach', 'Deployment'])

        # Check if required columns are found
        missing_columns = []
        if not ticket_col:
            missing_columns.append("TicketNumber")
        if not implementer_col:
            missing_columns.append("ImplementerGroup")
        if not deployment_col:
            missing_columns.append("Deployment Approach")

        if missing_columns:
            return {
                "status": "error",
                "error_message": f"Required columns not found: {', '.join(missing_columns)}",
                "available_columns": original_fieldnames,
                "suggestion": "Please ensure your file has columns: TicketNumber, ImplementerGroup, Deployment Approach (case-insensitive and space-agnostic matching)."
            }

        classification_results = []
        auto_count = 0
        hybrid_count = 0
        manual_count = 0
        unclassified_count = 0

        print(f"✅ Found required columns:")
        print(f"   📋 TicketNumber: '{ticket_col}'")
        print(f"   👥 ImplementerGroup: '{implementer_col}'")
        print(f"   🚀 Deployment Approach: '{deployment_col}'\n")

        total_crs = 0
        # Iterate over DataFrame rows
        for row_num, row_series in df.iterrows():
            total_crs += 1
            # row_num from iterrows is 0-indexed, add 2 to match original function's 1-indexed (header + 1)
            original_row_index = row_num + 2

            ticket_number = str(row_series.get(ticket_col, '')).strip()
            implementer_group = str(row_series.get(implementer_col, '')).strip()
            deployment_approach = str(row_series.get(deployment_col, '')).strip()

            # Skip empty rows (already handled by df.dropna(how='all'), but keeping this for robustness
            # in case some columns are present but others are empty. This check is more specific to the core columns)
            if not ticket_number and not implementer_group and not deployment_approach:
                total_crs -= 1 # Revert count if row is truly empty based on key columns
                continue

            cr_classification = classify_single_cr(
                ticket_number, implementer_group, deployment_approach, original_row_index, row_series.to_dict()
            )
            classification_results.append(cr_classification)

            # Count classifications
            cr_type = cr_classification["classification"].lower()
            if cr_type == "auto":
                auto_count += 1
            elif cr_type == "hybrid":
                hybrid_count += 1
            elif cr_type == "manual":
                manual_count += 1
            else:
                unclassified_count += 1

        output_directory = os.path.join(os.path.expanduser("~"), "Downloads")
        output_file = "results_file.csv"
        output_filepath = os.path.join(output_directory, output_file)

        output_result = create_output_dataframe_with_validation(
            df=df,
            classification_results=classification_results,
            output_filepath=output_filepath
        )

        if output_result is None:
            print(f"Output saved successfully to {output_filepath}")
        else:
            # This branch would execute if output_filepath was not provided
            print("Output content returned as string (should not happen if output_filepath is given).")

        return {
            "status": "success",
            "found_columns": {
                "ticket_number": ticket_col,
                "implementer_group": implementer_col,
                "deployment_approach": deployment_col
            },
            "summary": {
                "total_crs": total_crs,
                "auto_crs": auto_count,
                "hybrid_crs": hybrid_count,
                "manual_crs": manual_count,
                "unclassified_crs": unclassified_count,
                "classification_rate": f"{((total_crs - unclassified_count) / total_crs * 100):.1f}%" if total_crs > 0 else "0%"
            },
            "detailed_results": classification_results[:25],  # Show first 25 for readability
            "total_results_count": len(classification_results)
        }

    except Exception as e:
        return {
            "status": "error",
            "error_message": f"Error processing file content: {str(e)}"
        }

def find_column(fieldnames: List[str], possible_names: List[str]) -> str:
    """Find column name from possible variations (case-insensitive)."""
    for possible_name in possible_names:
        for fieldname in fieldnames:
            if possible_name.lower().replace(' ', '').replace('_', '') == fieldname.lower().replace(' ', '').replace(
                    '_', ''):
                return fieldname
    return ""

def classify_single_cr(ticket_number: str, implementer_group: str, deployment_approach: str, row_num: int, row: dict) -> dict:
    """
    Classifies a single CR based on ImplementerGroup and Deployment Approach.

    Args:
        ticket_number (str): Ticket number/ID
        implementer_group (str): Implementer group(s)
        deployment_approach (str): Deployment approach(es)
        row_num (int): Row number for reference

    Returns:
        dict: Classification result for the single CR
    """
    result = {
        "row_number": row_num,
        "ticket_number": ticket_number or f"Row_{row_num}",
        "implementer_group": implementer_group,
        "deployment_approach": deployment_approach,
        "classification": "Unclassified",
        "confidence": "Low",
        "reasoning": [],
        "issues": [],
        "validation": []
    }

    result['validation'] = validation_checks(row)

    # Check for missing data
    if not implementer_group:
        result["issues"].append("❌ ImplementerGroup is empty")
        result["reasoning"].append("Cannot classify without ImplementerGroup")
        return result

    if not deployment_approach:
        result["issues"].append("❌ Deployment Approach is empty")
        result["reasoning"].append("Cannot classify without Deployment Approach")
        return result

    # Parse implementer groups and deployment approaches
    groups = [g.strip() for g in implementer_group.split(',') if g.strip()]
    approaches = [a.strip() for a in deployment_approach.split(',') if a.strip()]

    # Check for specific patterns
    has_apprlse = "PSG_DBSAPP_APPRLSE" in groups
    other_psg_groups = [g for g in groups if re.match(r'PSG_DBSAPP_\w+', g) and g != "PSG_DBSAPP_APPRLSE"]

    # Check deployment approaches
    has_cicd = any("CICD" in approach or "CI/CD" in approach for approach in approaches)
    has_manual = any("Manual" in approach for approach in approaches)

    # Classification logic
    if (has_apprlse and not other_psg_groups and
            len(groups) == 1 and has_cicd and not has_manual):
        # AUTO CR
        result["classification"] = "Auto"
        result["confidence"] = "High"
        result["reasoning"] = [
            "✅ Single implementer group: PSG_DBSAPP_APPRLSE",
            "✅ Deployment approach contains CICD",
            "✅ No manual deployment mentioned"
        ]

    elif (has_apprlse and other_psg_groups and has_cicd and has_manual):
        # HYBRID CR
        result["classification"] = "Hybrid"
        result["confidence"] = "High"
        result["reasoning"] = [
            "✅ Has PSG_DBSAPP_APPRLSE group",
            f"✅ Has additional PSG groups: {', '.join(other_psg_groups)}",
            "✅ Has both CICD and Manual deployment approaches"
        ]

    elif (not has_apprlse and other_psg_groups and
          not has_cicd and has_manual):
        # MANUAL CR
        result["classification"] = "Manual"
        result["confidence"] = "High"
        result["reasoning"] = [
            f"✅ Implementer group: {', '.join(other_psg_groups)}",
            "✅ No PSG_DBSAPP_APPRLSE group",
            "✅ Manual deployment approach",
            "✅ No CICD deployment mentioned"
        ]

    else:
        # UNCLASSIFIED - provide detailed reasoning
        result["classification"] = "Unclassified"
        result["confidence"] = "Low"
        result["reasoning"] = ["❓ Does not match standard CR patterns"]

        # Detailed analysis for unclassified
        if not any(re.match(r'PSG_DBSAPP_\w+', g) for g in groups):
            result["issues"].append("❌ No valid PSG_DBSAPP_* groups found")

        if not (has_cicd or has_manual):
            result["issues"].append("❌ No recognized deployment approach (CICD/Manual)")

        # Suggest possible classification
        if has_apprlse:
            if other_psg_groups:
                result["reasoning"].append("💡 Might be Hybrid - has multiple PSG groups")
            else:
                result["reasoning"].append("💡 Might be Auto - has PSG_DBSAPP_APPRLSE")
        elif other_psg_groups:
            result["reasoning"].append("💡 Might be Manual - has specific PSG group")

    return result

def _get_validation_summary(cr_result: Optional[Dict[str, Any]]) -> str:
    if not cr_result:
        return "No classification data available."

    summary_parts = []

    validation_details = []
    validations = cr_result.get('validation', [])

    for check in validations:
        # Assuming validation results from validation_checks are strings like "Checking X ---> Correct format"
        # or "Checking Y ---> Update Z"
        if isinstance(check, str) and check.strip():
            validation_details.append(check.strip())

    if validation_details:
        validation_text = "; ".join(validation_details)
        summary_parts.append(f"Validation: {validation_text}")

    if summary_parts:
        return "; ".join(summary_parts)
    else:
        # Fallback if no issues or validation details, but cr_result exists
        return "No validation issues detected"

def create_output_dataframe_with_validation(
    df: pd.DataFrame,
    classification_results: List[Dict[str, Any]],
    output_filepath: Optional[str] = None
) -> Optional[str]:
    """
    Creates an output file content string by adding a 'Validation_Results' column
    to the input DataFrame.

    Args:
        df (pd.DataFrame): The input pandas DataFrame containing the original data.
        classification_results (List[Dict[str, Any]]): A list of dictionaries, where each
                                                      dictionary contains classification
                                                      and validation results for a CR.
        output_format (str): The desired output format ('csv', 'tsv', 'txt'). Defaults to 'csv'.
        delimiter (str): The delimiter to use for the output string if format is 'csv', 'tsv', or 'txt'.
                         Defaults to ','. Ignored for other formats (though currently only CSV/TSV/TXT are supported).

    Returns:
        str: The content of the new file with an added 'Validation_Results' column, as a string.
    """
    if df.empty:
        raise ValueError("Input DataFrame cannot be empty.")

    output_delimiter = ','

    results_map = {res['row_number'] - 2: res for res in classification_results if res.get('row_number')}
    # The key in results_map will now correspond to the 0-indexed DataFrame row index.

    validation_column_name = "Validation_Results"

    # Create a new column in the DataFrame for validation results
    # Initialize with empty string to avoid NaNs if no result is found for a row
    df_output = df.copy() # Work on a copy to avoid modifying the original DataFrame in place
    df_output[validation_column_name] = ''

    # Populate the new column
    for index, row_series in df_output.iterrows():
        cr_result = results_map.get(index) # Use the 0-indexed DataFrame index
        validation_summary = _get_validation_summary(cr_result)
        df_output.at[index, validation_column_name] = validation_summary

    # Convert the DataFrame to a string based on the specified output format
    if output_filepath:
        timestamp = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")

        # Split the filepath to insert timestamp before extension
        file_dir = os.path.dirname(output_filepath)
        file_name = os.path.basename(output_filepath)
        name, ext = os.path.splitext(file_name)

        timestamped_filename = f"{name}_{timestamp}{ext}"
        timestamped_filepath = os.path.join(file_dir, timestamped_filename)

        if os.path.exists(timestamped_filepath):
            os.remove(timestamped_filepath)
        df_output.to_csv(timestamped_filepath, sep=output_delimiter, index=False, encoding='utf-8')
        return None  # File written directly
    else:
        output_buffer = io.StringIO()
        df_output.to_csv(output_buffer, sep=output_delimiter, index=False)
        return output_buffer.getvalue()

class validation_error(Exception):

def check_ChangeNature_and_CodeChange(row: dict) -> str:
    changeNature = str(row.get('ChangeNature', '')).strip().lower()
    codeChange = str(row.get('Any application / programme code change', '')).strip().lower()
    check_name = "Checking ChangeNature and CodeChange"

    if changeNature == 'business enhancement' and codeChange in ['n', 'no']:
        raise validation_error(f"{check_name} ---> Update 'Any application / programme code change' to YES OR update ChangeNature to 'System/Business recovery & support, capacity upgrades'")

    elif "system/business" in changeNature and 'y' in codeChange:
        raise validation_error(f"{check_name} ---> Update 'Any application / programme code change' to NO OR update ChangeNature to 'Business Enhancement'")

    else:
        return f"{check_name} ---> Correct format"

def check_Location_and_ChangeSummary(row: dict) -> str:
    location = str(row.get('Location', '')).strip().lower()
    changeSummary = set(str(row.get('ChangeSummary', '')).strip().lower().split())
    check_name = "Checking Location and ChangeSummary"

    environments = [
        'DR',
        'PAT',
        'Pre-prod',
        'Prod',
        'Prod and DR',
        'QA'
    ]

    for env in environments:
        env_lower = env.lower()
        env_words = set(env_lower.split())

        if env_words.issubset(changeSummary):
            if env_lower == location:
                return f"{check_name} ---> Correct format"
            else:
                raise validation_error(f"{check_name} ---> Location ({location}) and ChangeSummary ({env.lower()}) should have similar environments.")

    return f"{check_name} ---> Correct format"

def check_CodeChange_and_ApplicationImpacted(row: dict) -> str:
    codeChange = str(row.get('Any application / programme code change', '')).strip().lower()

    appCodes = set([code.strip() for code in appCodes])

    check_name = "Checking CodeChange and ApplicationImpacted"

    num_apps = len(appCodes) - 1 if 'RAF' in appCodes else len(appCodes)
    if codeChange == 'yes - more than 3 apps':
        if num_apps < 3:
            raise validation_error(f"{check_name} ---> Number of App Codes mentioned in 'ApplicationImpacted' is less than 3 but 'Any application / programme code change' mentions more than 3 apps.")
        else:
            crClassification = str(row.get('CR classification', '')).strip().lower()

                raise validation_error(f"{check_name} ---> Update CR Classification to 'Heightened'")

    elif codeChange in ['y', 'yes', 'yes - not more than 3 apps']:
        if num_apps > 3:
            raise validation_error(f"{check_name} ---> Number of App Codes mentioned in 'ApplicationImpacted' is more than 3 but 'Any application / programme code change' mentions less than 3 apps.")

    return f"{check_name} ---> Correct format"

def check_DownTime_Window(row: dict) -> str:
    downTime_start = str(row.get('DowntimeStartDateTime', '')).strip()
    downTime_end = str(row.get('DowntimeEndDateTime', '')).strip()
    check_name = "Checking DownTime Window"

    if downTime_start == 'nan' and downTime_end == 'nan':
        return f"{check_name} ---> Correct format"

    if (downTime_start == 'nan') != (downTime_end == 'nan'):
        raise validation_error(f"{check_name} ---> Both start and end times must be provided or both empty")

    try:
        downtime_start_dt = datetime.strptime(downTime_start, '%d/%m/%Y %H:%M')
        downtime_end_dt = datetime.strptime(downTime_end, '%d/%m/%Y %H:%M')

        downtime_duration = downtime_end_dt - downtime_start_dt

        if downtime_duration < timedelta(0):
            raise validation_error(f"{check_name} ---> DownTime duration can't be less than 0")
        if downtime_duration > timedelta(hours=3):
            raise validation_error(f"{check_name} ---> Why was DownTime window greater than 3 hrs? ")

    except (ValueError, TypeError):
        raise validation_error(f"{check_name} ---> Couldn't parse datetime string.")

    return f"{check_name} ---> Correct format"

def check_version_upgrade_format(text):
    version_pattern = r"v?(\d+(\.\d+){0,4}(?:[a-zA-z][\w.-]*)?)"

    complete_pattern = re.compile(
        r"(?:version\s+upgrade|upgrade\s+version|upgrade)\s+"
        r".*?"
        r"(?:from|frm)\s+"
        r".*?" + version_pattern + r"\s+"
        r".*?"
        r"(?:to|2)\s+"
        r".*?" + version_pattern,
        re.IGNORECASE
    )

    general_upgrade_mention_pattern = re.compile(
        r"(?:version\s+upgrade|upgrade\s+version|upgrade)",
        re.IGNORECASE
    )

    match_complete = complete_pattern.search(text)
    if match_complete:
        from_version = match_complete.group(1)
        to_version = match_complete.group(4)
        return True, from_version, to_version

    else:
        match_general = general_upgrade_mention_pattern.search(text)
        if match_general:
            return True, None, None
        else:
            return False, None, None

def check_changeSummary_for_versionUpgrade(row: dict) -> str:
    changeSummary = str(row.get('ChangeSummary', '')).lower()
    changeSummary = [
        summary.strip() for summary in re.split(r"\.(?:\s+|\n+)", changeSummary) if summary.strip()
    ]
    check_name = "Checking Change Summary For Version Upgrade"

    found = False
    for summary in changeSummary:
        found, from_version, to_version = check_version_upgrade_format(summary)
        if found:
            if from_version is not None and to_version is not None:
                continue
            else:
                raise validation_error(f"{check_name} ---> Version Upgrade not found. Please check.")
        else:
            continue

    return f"{check_name} ---> Correct format"

def validation_checks(row: dict) -> list:

    errors =[]
    statements = []

    checks_to_perform = [
        check_ChangeNature_and_CodeChange,
        check_Location_and_ChangeGroup,
        check_ChangeNature_and_ChangeSummary,
        check_ChangeSummary_and_ChangeNature,
        check_Location_and_ChangeSummary,
        check_CR_Approval_Status,
        check_AppResiliency_and_CRClassification,
        check_CodeChange_and_ApplicationImpacted,
        check_ReversionApproach,
        check_UAT_signoff_status,
        checking_DeploymentApproach,
        check_CyberArk_Objects,
        check_ApplicationImpacted,
        check_DownTime_Window,
        check_changeSummary_and_changeDescription,
        check_changeSummary_for_versionUpgrade
    ]

    for check in checks_to_perform:
        try:
            result = check(row)
            statements.append(result)
        except validation_error as e:
            errors.append(e)
            statements.append(str(e))
        except Exception as e:
            print("Unexpected error occurred:", e)

    if len(errors) > 0:
        statements.append("Total validation issues: {} ❌\n".format(len(errors)))
    else:
        statements.append("No validation issues found ✅\n")

    return statements
